591  def _make_full_seq_to_type(self, seq_to_type):
592  """!
593  @param seq_to_type
594  A dictionary from sequences of atom types to types of topology
595  components consisting of those atoms.
596 
597  @return
598  A dictionary from sequences of atom types to types of topology
599  components consisting of those atoms. This dictionary is
600  different from the `seq_to_type` by containing sequences
601  in both directions as its keys.
602 
603  This method overrides LammpsTopology._make_full_seq_to_type
604  to be compatible with impropers.
605  Different from other topology components
606  where atoms are connected *linearly*, impropers have a center atom
607  to which all the other three atoms connect. For example,
608  sequence `(1, 2, 3, 4)`, `(1, 2, 4, 3)`, `(1, 3, 2, 4)`,
609  `(1, 3, 4, 2)`, `(1, 4, 2, 3)` and `(1, 4, 3, 2)` are reduced
610  to one improper type, where the first atom in the sequences is
611  the center atom of improper.
612  """
613 
614  c = 1 if self._class2 else 0
615 
616  return {
617  t[:c] + (k[c],) + t[c:]: v
618  for k, v in seq_to_type.items()
619  for t in it.permutations(k[:c]+k[c+1:])
