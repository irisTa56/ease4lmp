411  for k in bonds_per_atom[j]:

413  continue

415  for l in bonds_per_atom[k]:
416  if l == j or l == i:
417  continue
418  self._data[i]["1-4"].append(l+1)

421  return max([sum([len(v) for v in d.values()]) for d in self._data])
