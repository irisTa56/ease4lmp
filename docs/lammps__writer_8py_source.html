191  """!
192  This method write Lammps DATA file.

194  [Arguments]
195  * path: <str>

199  2. 'num_*_type'
200  """
201 
202  num_atom = self._lmp_atoms.get_num()

204  if "num_atom_type" in kwargs else self._lmp_atoms.get_num_type()

206  num_topo = {
207  k: v.get_num() if v.get_num_type() else 0
208  for k, v in self._topo.items()} \

211  if self._topo else dict.fromkeys(topo_keys, 0)
212 
213  for k in topo_keys:
214  name = "num_{}_type".format(k)
215  if name in kwargs:
216  num_topo_type[k] = kwargs[name]
217 
218  with open(path, "w") as f:
219 
220  f.write("# written by ease4lmp.LammpsWriter at {}\n".format(
221  dt.datetime.now().strftime("%Y/%m/%d %H:%M:%S")))
222 
223  f.write("\n{} atoms\n".format(num_atom))
224  f.write("".join([
225  "{} {}s\n".format(num_topo[k], k) for k in topo_keys]))
226 
227  f.write("\n{} atom types\n".format(num_atom_type))

229  "{} {} types\n".format(num_topo_type[k], k) for k in topo_keys]))
230 
231  xlo, ylo, zlo, = 0.0, 0.0, 0.0

233 
234  if centering:

240  ylo = -yhi
241  zlo = -zhi

244 
245  f.write("\n{}".format("".join([
246  "{0:.8e} {1:.8e} {2}lo {2}hi\n".format(lo, hi, x)
247  for lo, hi, x in zip(
248  [xlo, ylo, zlo], [xhi, yhi, zhi], ["x", "y", "z"])])))
249 
250  if self._prism.is_skewed():
251  f.write("\n{:.8e} {:.8e} {:.8e} xy xz yz\n".format(
252  *map(float, (xy, xz, yz))))
253 
254  if mass:

256
