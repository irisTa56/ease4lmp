50  def add_bond(self, atom1, atom2, img1=(0,0,0), img2=(0,0,0)):
51  """!
52  This method add a bond connecting two atoms (specified by index).
53  The bond data is stored as a property of the first atom and the
54  second atoms as following.
55 
56  [
57  [rel-id, rel-img_x, rel-img_y, rel-img_z], # 1st bond
58  [rel-id, rel-img_x, rel-img_y, rel-img_z], # 2nd bond
59  ...
60  ]
61 
62  'rel-id' is a relative id; id of the second atom minus id of the
63  first atom, for the first atom. 'rel-img's is relative image flags
64  used in the case of periodic boundary conditions.
65 
66  [Arguments]
67  * atom1: <int>
68  * atom2: <int>
69  * img1: <tuple/list>
70  * img2: <tuple/list>
71  """
72 
73  if not isinstance(img1, (tuple, list)) or len(img1) != 3:
74  raise RuntimeError(
75  "Image flag 'img1' should be 3-membered tuple/list")
76  elif not isinstance(img2, (tuple, list)) or len(img2) != 3:
77  raise RuntimeError(
78  "Image flag 'img2' should be 3-membered tuple/list")
79 
80  relative_data = np.insert(
81  np.array(img2, int) - np.array(img1, int), 0, atom2 - atom1)
82 
83  idx1 = self._get_available_bond_index(atom1, relative_data)
84  idx2 = self._get_available_bond_index(
85  atom2, -relative_data, 1 if atom1 == atom2 else 0)
86 
87  self.arrays["bonds"][atom1][idx1] = relative_data
88  self.arrays["bonds"][atom2][idx2] = -relative_data

90  def change_max_bonds(self, n=4):
91  """!
92  This method change the maximum number of bonds per atom. Default is
93  four bonds per atom.
94 
95  [Arguments]
96  * n: <int>; new maximum number of bonds per atom
97  """
98 
99  if n < self._max_bonds:
100  self.arrays["bonds"] = np.array([
101  bs[:n] for bs in self.arrays["bonds"]])
102  elif self._max_bonds < n:
103  zeros = np.zeros((n - self._max_bonds, 4), int)
104  self.arrays["bonds"] = np.array([
105  np.append(bs, zeros, axis=0) for bs in self.arrays["bonds"]])
106 
107  self._max_bonds = n

249  def __delitem__(self, idx):
250  """!
251  [Arguments]
252  * idx: <tuple/list/int>
253  """
254 
255  if isinstance(idx, int):
256  idx = np.array([idx])
257  elif isinstance(idx, (tuple, list)) and len(idx) > 0:
258  idx = np.array(idx)
259 
260  mask = np.ones(len(self), bool)
261  mask[idx] = False
262 
263  for i, bs in enumerate(self.arrays["bonds"]):
264

280  def __imul__(self, m):
281  """!
282  In-place repeat of atoms.
283 
284  [Arguments]
285  * m: <tuple/list/int>
286  [Return]
287  <BondedAtoms>
288  """
289 
290  if isinstance(m, int):
291  m = (m, m, m)
292 
293  n = len(self)
294 
295  super().__imul__(m)
296 
297  bonds = self.arrays["bonds"]
298  i0 = 0
299 
300  for m0 in range(m[0]):
301  for m1 in range(m[1]):
302  for m2 in range(m[2]):
303  i1 = i0 + n
304  for i in range(i0, i1):
305  bs = [b for b in bonds[i] if np.any(b[1:] != 0)]
306  for b in bs:
307  img_old = b[1:]
308  img_new = np.array([
309  (m0 + img_old[0]) // m[0],
310  (m1 + img_old[1]) // m[1],
311  (m2 + img_old[2]) // m[2]], int)
312  tmp = img_old - img_new * m # m will be converted to <numpy.ndarray>
313  b[0] += n * (tmp[0]*m[1]*m[2] + tmp[1]*m[2] + tmp[2])
314  b[1:] = img_new
315  i0 = i1
316 
317  return self

319  def _get_available_bond_index(self, atom, relative_data, offset=0):
320  """!
321  [Arguments]
322  * atom: <int>
323  * relative_data: <numpy.ndarray>
324  * offset: <int>; postpone returning index until this becomes zero
325  [Return]
326  <int>
327  """
328 
329  for i, b in enumerate(self.arrays["bonds"][atom]):
330  if (b == relative_data).all():
331  raise RuntimeError(
332  "Bond between atom '{}' and '{}' already exists".format(
333  atom, atom+relative_data[0]))
334  elif np.all(b == 0):
335  if offset == 0:
336  return i
337  else:
338  offset -= 1
339 
340  raise RuntimeError(
341  "Too many bonds attache to atom '{}'".format(atom))
