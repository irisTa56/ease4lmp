16 class BondedAtoms(ase.Atoms):
17  """!
18  This class inherits from `ase.Atoms` class, and has functionalities
19  to deal with (covalent) bonds connecting a pair of atoms.
20 
21  An instance of this class has bond data as a `numpy.ndarray`
22  as with other atomic properties like positions and velocities.
23  Shape of the array for bond data is (*N*, #_max_bonds, 4),
24  where *N* is the number of atoms. The first axis of the array
25  corresponds to atoms, and the second axis corresponds to
26  bonds connected to each atom. The third axis contains data for
27  each bond: an integer at 0 is relative index (of first axis)
28  for the other atom, integers from 1 to 3 are relative image flags
29  (to resolve a periodic boundary condition) for the other atom
30  in the *x*, *y* and *z* direction, respectively.
31 
32  Basically, you can use methods of `ase.Atoms` in the same manner.
33  """

90  a periodic simulation box the second atom is in.

107  np.array(img2, int) - np.array(img1, int), 0, atom2 - atom1)

110  atom1, relative_data)

357  def __imul__(self, m):
358  """!
359  @param m
360  Three-element tuple or list defining how many times
361  a periodic simulation box is repeated in each direction.
362  The first, second and third element corresponds to the *x*, *y*
363  and *z* direction, respectively.
364  If an integer is given as `m`, a tuple `(m, m, m)` is used.
365 
366  @return
367  A BondedAtoms instance containing all the repeated atoms
368  with the expanded simulation box.
369 
370  This method overrides `ase.Atoms.__imul__`.
371  After calling the parent method to expand the simulation box,
372  this method resolves bonds originally connecting two atoms in
373  different images of the periodic simulation box.
374  """
375 
376  if isinstance(m, int):
377  m = (m, m, m)
378 
379  n = len(self)
380 
381  super().__imul__(m)
382 
383  bonds = self.arrays["bonds"]
384  i0 = 0
385 
386  for m0 in range(m[0]):
387  for m1 in range(m[1]):
388  for m2 in range(m[2]):
389  i1 = i0 + n
390  for i in range(i0, i1):
391  bs = [b for b in bonds[i] if np.any(b[1:] != 0)]
392  for b in bs:
393  img_old = b[1:]
394  img_new = np.array([
395  (m0 + img_old[0]) // m[0],
396  (m1 + img_old[1]) // m[1],
397  (m2 + img_old[2]) // m[2]], int)
398  tmp = img_old - img_new * m # m will be converted to <numpy.ndarray>
399  b[0] += n * (tmp[0]*m[1]*m[2] + tmp[1]*m[2] + tmp[2])
400  b[1:] = img_new
401  i0 = i1
402 
403  return self
